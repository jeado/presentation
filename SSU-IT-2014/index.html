function makeDropBox(dropbox, listData){
 var option;
	
 if(listData){
  for(var i = 0, length = listData.length; i < length; i++) {
   option = document.createElement("option");
   option.value = listData[i].cdDetailNo;
   option.text = listData[i].cdDetailName;
   dropbox.options.add(option);
  }
 }

function setGridData(listData){
 var list = "";

 if(listData){
  for(var i = 0, length = listData.length; i < length; i++) {
   list += "<tr>";
   for(var j = 0, length2 = gridColumn.length; j < length2; j++) {
    list += "<td>";
    list += listData[i][gridColumn[j]] || " ";
    list += "</td>";
   }
   list += "</tr>";
  }
}

 gridBody.innerHTML = list; // 값 변경

function userMgtCtr ($scope,$http) {
 $scope.sObj={deptId:'',posId:''};
 $scope.deptList = [
  { deptId: '001', deptNm: '부서A'},
  { deptId: '002', deptNm: '부서B'}
 ];

 $scope.posList = [
  { posId: '001', posNm: '사원'},
  { posId: '002', posNm: '대리'}
 ];

 $scope.find = function (sObj) {
  $http({
   method: 'GET', url: 'user.json',
   params: { deptId: sObj.deptId, posId: sObj.posId }
  })
   .success(function(data, status, headers, config) {
    $scope.userList = data;
   });
 };
}

Directive 제공

directive?

지시자란?


directives are markers on a DOM element (such as an attribute, element name, or CSS class) that tell AngularJS's HTML compiler ($compile) to attach a specified behavior to that DOM element or even transform the DOM element and its children. (from AngularJS Officail DOC)
